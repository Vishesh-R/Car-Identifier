function updateOtherLabels(predictions) { labelOtherContainer = document.getElementById("other-container"); labelOtherContainer.innerHTML = ""; let highestProbability = -1; let secondHighestProbability = -1; let highestClassIndex = -1; let secondHighestClassIndex = -1; // Find the class with the highest and second-highest probability for (let i = 0; i < predictions.length; i++) { if (predictions[i].probability > highestProbability) { secondHighestProbability = highestProbability; secondHighestClassIndex = highestClassIndex; highestProbability = predictions[i].probability; highestClassIndex = i; } else if (predictions[i].probability > secondHighestProbability) { secondHighestProbability = predictions[i].probability; secondHighestClassIndex = i; } } // Sort predictions alphabetically by className predictions.sort((a, b) => a.className.localeCompare(b.className)); // Recompute sorted indices for highest and second-highest const sortedHighestIndex = predictions.findIndex(p => p.probability === highestProbability); const sortedSecondHighestIndex = predictions.findIndex(p => p.probability === secondHighestProbability); // Display all predictions with highlighting the highest and second-highest for (let i = 0; i < predictions.length; i++) { const classPrediction = predictions[i].className + " : " + (predictions[i].probability.toFixed(3) * 100) + "%"; const div = document.createElement("div"); div.innerHTML = classPrediction; // Highlight the highest probability class if (i === sortedHighestIndex) { div.classList.add("highlight"); } // Highlight the second-highest probability class else if (i === sortedSecondHighestIndex) { div.classList.add("highlight-second"); } labelOtherContainer.appendChild(div); } } // Load Excel data initially loadExcelFile();
